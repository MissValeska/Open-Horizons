78  };
79

81  if (ch == '[') break; // Skip sections
82 
83  // Parse option name
84  while (((ch >= 'A') && (ch <= 'Z')) ||
85  ((ch >= 'a') && (ch <= 'z')) ||

90  }

92 
93  // Skip whitespace

102

106  nextline:
